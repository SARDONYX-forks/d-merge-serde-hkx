Options to customize writing floats.

ยงExamples

use lexical_write_float::Options;

let options = Options::builder()
    .trim_floats(true)
    .nan_string(Some(b"NaN"))
    .inf_string(Some(b"Inf"))
    .build()
    .unwrap();
