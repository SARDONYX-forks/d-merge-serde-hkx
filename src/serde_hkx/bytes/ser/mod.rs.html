25/// To hkx binary file data.
26///
27/// # Errors
28/// - When information necessary for binary data conversion is missing.
29/// - When a write to the wrong write position is requested.
30///
31/// # NOTE
32/// - To reproduce the original hkx, it is necessary to arrange them in dependency order using `HavokSort::sort_for_bytes` or similar.
33/// - If you convert in the order of hkx => XML => hkx, you may not be able to restore the original hkx due to data loss during XML conversion.
34pub fn to_bytes<V>(value: &V, header: &HkxHeader) -> Result<Vec<u8>>
35where
36    V: Serialize + ClassNamesWriter,
37{
38    to_bytes_with_opt(
39        value,
40        header,
41        ByteSerializer {
42            is_little_endian: match header.endian {
43                0 => false, // big endian
44                1 => true,  // little endian
45                invalid => InvalidEndianSnafu { invalid }.fail()?,
46            },
47            is_x86: match header.pointer_size {
48                4 => true,
49                8 => false,
50                invalid => UnsupportedPtrSizeSnafu { invalid }.fail()?,
51            },
52            ..Default::default()
53        },
54    )
55}

57/// Serialize to hkx bytes with `event`&`variable` ID maps + header.
58///
59/// # Errors
60/// Returns an error when
61/// - When ptr_size is other than 4/8
62/// - When endian is other than 0/1
63#[inline]
64pub fn to_bytes_with_maps<V>(
65    value: &V,
66    header: &HkxHeader,
67    event_id_map: EventIdMap,
68    variable_id_map: VariableIdMap,
69) -> Result<Vec<u8>, Error>
70where
71    V: Serialize + ClassNamesWriter,
72{
73    let ser = ByteSerializer {
74        event_id_map,
75        variable_id_map,
76        is_little_endian: match header.endian {
77            0 => false, // big endian
78            1 => true,  // little endian
79            invalid => InvalidEndianSnafu { invalid }.fail()?,
80        },
81        is_x86: match header.pointer_size {
82            4 => true,
83            8 => false,
84            invalid => UnsupportedPtrSizeSnafu { invalid }.fail()?,
85        },
86        ..Default::default()
87    };
88
89    to_bytes_with_opt(value, header, ser)
90}

92/// Serialize to bytes with custom `BytesSerializer` settings.
93///
94/// # Errors
95/// - When information necessary for binary data conversion is missing.
96/// - When a write to the wrong write position is requested.
97pub fn to_bytes_with_opt<V>(value: &V, header: &HkxHeader, ser: ByteSerializer) -> Result<Vec<u8>>
98where
99    V: Serialize + ClassNamesWriter,
100{
101    let mut serializer = ser;
102
103    // 1/5: root header
104    serializer.output.write_all(&header.to_bytes())?;
105    serializer.contents_class_name_section_index = header.contents_class_name_section_index;
106    serializer.contents_section_index = header.contents_section_index;
107
108    // 2/5: Get Section headers positions.(because the values of the fixups are not yet known.)
109    let classnames_header_start = 64 + header.padding_size() as u64; // 64: Root header size
110    let types_header_start = classnames_header_start + 48; // next `SectionHeader`(size 48bytes) position.
111    let data_header_start = types_header_start + 48;
112    serializer.output.set_position(data_header_start + 48);
113
114    // 3/5: section contents
115    // - `__classnames__` section
116    if serializer.is_little_endian {
117        serializer.class_starts =
118            value.write_classnames_section::<LittleEndian>(&mut serializer.output)?;
119    } else {
120        serializer.class_starts =
121            value.write_classnames_section::<BigEndian>(&mut serializer.output)?;
122    };
123    #[cfg(feature = "tracing")]
124    tracing::trace!("class_starts: {:#?}", serializer.class_starts);
125
126    // - `__data__` section
127    serializer.abs_data_offset = header.padding_size() + serializer.output.position() as u32;
128    value.serialize(&mut serializer)?;
129
130    // 4/5: Write fixups_offsets of `__data__` section header.
131    serializer.output.zero_fill_align(16)?; // Always make the start of fixups a multiple of 16.
132    let (local_offset, global_offset, virtual_offset) = serializer.write_data_fixups()?; // Write local, global and virtual fixups
133    let exports_offset = serializer.relative_position()?; // This is where the exports_offset is finally obtained.
134
135    // 5/5 Write remain offsets for `__types__` & `__data__` section header.
136    let abs_data_start = serializer.abs_data_offset;
137    let data_section_header = SectionHeader {
138        section_tag: SectionHeader::DATA_SECTION_HEADER_TAG,
139        section_tag_separator: 0xff,
140        absolute_data_start: abs_data_start,
141        local_fixups_offset: local_offset,
142        global_fixups_offset: global_offset,
143        virtual_fixups_offset: virtual_offset,
144        exports_offset,
145        imports_offset: exports_offset,
146        end_offset: exports_offset,
147    };
148    #[cfg(feature = "tracing")]
149    tracing::trace!("data_section_header = {data_section_header}");
150
151    serializer.output.set_position(classnames_header_start);
152    let section_offset = header.section_offset;
153
154    if serializer.is_little_endian {
155        tri!(serializer.write_section_headers::<LittleEndian>(
156            section_offset,
157            types_header_start,
158            data_header_start,
159            &data_section_header,
160        ));
161    } else {
162        tri!(serializer.write_section_headers::<BigEndian>(
163            section_offset,
164            types_header_start,
165            data_header_start,
166            &data_section_header,
167        ));
168    };
169
170    Ok(serializer.output.into_inner())
171}

173/// Binary data serializer
174///
175/// # Note
176/// - All of these fixups are from the `__data__` section.
177#[derive(Debug, Default)]
178pub struct ByteSerializer<'ser> {
179    /// Endianness of serialization target
180    is_little_endian: bool,
181    /// Ptr size of serialization target.
182    /// 32bit: x86
183    is_x86: bool,
184
185    /// Bytes
186    output: Cursor<Vec<u8>>,
187
188    /// `eventNames` indexes of `hkbBehaviorGraphStringData`.
189    ///
190    /// # Example
191    /// If the nemesis id e.g. `$eventID[variableSample]$`, then `variableSample` is the key,
192    ///
193    /// The value is the index of the `eventSample` element in the `variableNames` field array of the `hkbBehaviorGraphStringData` class.
194    ///
195    /// # Note
196    /// This ID substitution takes place only for integer types.
197    ///
198    /// The pointer refers to the special ID (`e.g. $aaa$10`) of Nemesis as it is.
199    event_id_map: EventIdMap<'ser>,
200    /// `variableNames` indexes of `hkbBehaviorGraphStringData`.
201    ///
202    /// # Example
203    /// If the nemesis id e.g. `$variableID[variableSample]$`, then `variableSample` is the key,
204    ///
205    /// The value is the index of the `variableSample` element in the `variableNames` field array of the `hkbBehaviorGraphStringData` class.
206    ///
207    /// # Note
208    /// This ID substitution takes place only for integer types.
209    ///
210    /// The pointer refers to the special ID (`e.g. $aaa$10`) of Nemesis as it is.
211    variable_id_map: VariableIdMap<'ser>,
212
213    /// This is cached to find the relative position of the binary.
214    abs_data_offset: u32,
215
216    /// Flag to branch special align16 conditions such as the following.
217    /// Switch only in `SerializeStruct::serialize_array` and read only in `serialize_cow`.
218    /// - `hkArray<CString>` / `hkArray<StringPtr>` -> align2 is done after string serialization.
219    /// - For a single field -> align16 is done after string serialization.
220    is_in_str_array: bool,
221
222    // ---- local fixup information
223    /// The position to which the pointer returns after writing the end of the pointer.
224    /// Holds positional information for writing the destination of the nested array pointers in the field,
225    /// where the Nth element means N hierarchically nested.
226    /// - Example: `u32` <- `ClassB.b: u32` <- `ClassA.a: Array<ClassB>` <- `Array<ClassA>`
227    pointed_pos: Vec<u64>,
228    /// Coordination information to associate a pointer of a pointer type of a field in a class with the data location to which it points.
229    ///
230    /// # Note
231    /// All of these fixups are from the DATA SECTION.
232    ///
233    /// The following are not recorded in `local_fixup`.
234    /// - If `Array<T>` is empty.
235    /// - `CString`, `StringPtr` points to null ptr.
236    local_fixups: Vec<u8>,
237
238    // ---- Global fixup information
239    /// A map that holds the src of global_fixups until the dst of virtual_fixups is known.
240    /// -   key: Starting point of the binary for which the pointer class write is requested.
241    /// - value: Unique class pointer.(e.g. XML: #0050 -> 50)
242    global_fixups_ptr_src: IndexMap<u32, Pointer<'static>>,
243    /// The `global_fixup.dst` == `virtual_fixup.src`.
244    ///
245    /// Therefore, the write start position must be retained.
246    /// The position of dst of global_fixup will be known after all the binary data of all classes are written.
247    /// - key: Unique class pointer.(e.g. XML: #0050 -> 50)
248    /// - value: Starting point where Havok Class binary data is written.
249    ///
250    /// # Note
251    /// - This is used as a key since no duplicate ptr is required at the same relative position.
252    ///   The ptr may be shared-referenced and cannot be keyed.
253    virtual_fixups_ptr_src: HashMap<Pointer<'static>, u32>,
254    /// Index of the contents section.(To write `global_fixups` index)
255    ///
256    /// It is usually `2` and refers to the index of `__data__`.
257    contents_section_index: i32,
258
259    // ---- Virtual fixup information
260    /// C++ Class constructor positions map binary temporally buffer.
261    ///
262    /// Finally, write to the data for output.
263    virtual_fixups: Vec<u8>,
264    /// This information is needed in `virtual_fixup.name_offset`.
265    ///
266    /// This is created by writing the `__classnames__` section.
267    /// - key: class name
268    /// - value: class name start position
269    class_starts: ClassStartsMap,
270    /// Index of the contents class name section.(To write virtual fixups index)
271    ///
272    /// It is usually `0` and refers to the index of `__classnames__`.
273    contents_class_name_section_index: i32,
274}

276impl<'ser> ByteSerializer<'ser> {
277    /// Get the position relative to the start of the `__data__` section.
278    #[inline]
279    fn relative_position(&self) -> Result<u32> {
280        let position = self.output.position() as u32;
281        let abs_data_offset = self.abs_data_offset;
282        position
283            .checked_sub(abs_data_offset)
284            .ok_or(Error::OverflowSubtractAbs {
285                position,
286                abs_data_offset,
287            })
288    }
289
290    /// Write `global_fixups` of data section bytes to writer.
291    ///
292    /// # Info
293    /// If all virtual_fixups are not obtained, references may not be available?
294    ///
295    /// # Note
296    /// `global_fixup.dst` == `virtual_fixup.src`
297    fn write_global_fixups(&mut self) -> Result<()> {
298        for (&src, ptr) in &self.global_fixups_ptr_src {
299            if let Some(&dst) = self.virtual_fixups_ptr_src.get(ptr) {
300                #[cfg(feature = "tracing")]
301                {
302                    let src_abs = self.abs_data_offset + src;
303                    let dst_abs = self.abs_data_offset + dst;
304                    tracing::debug!(
305                        "[global_fixups]({:#x}) src({src}:{src:#x}/abs {src_abs:#x}) -> {ptr} dst({dst}:{dst:#x}/abs: {dst_abs:#x})",
306                        self.output.position()
307                    );
308                }
309
310                if self.is_little_endian {
311                    self.output.write_u32::<LittleEndian>(src)?; // src
312                    self.output
313                        .write_i32::<LittleEndian>(self.contents_section_index)?; // dst_section_index
314                    self.output.write_u32::<LittleEndian>(dst)?; // dst(virtual_fixup.dst)
315                } else {
316                    self.output.write_u32::<BigEndian>(src)?; // src
317                    self.output
318                        .write_i32::<BigEndian>(self.contents_section_index)?; // dst_section_index
319                    self.output.write_u32::<BigEndian>(dst)?; // dst(virtual_fixup.dst)
320                }
321            } else {
322                return MissingGlobalFixupClassSnafu {
323                    ptr: ptr.to_static(),
324                }
325                .fail();
326            }
327        }
328        Ok(())
329    }
330
331    /// Write to temporary virtual_fixup data.
332    ///
333    /// # Info
334    /// Since the `class_name` and its location are known when the `__classnames__` section is written, the pair can be
335    /// written the moment virtual_fixup.src is available.
336    fn write_virtual_fixups_pair(
337        &mut self,
338        class_name: &'static str,
339        virtual_src: u32,
340    ) -> Result<()> {
341        if let Some(class_name_offset) = self.class_starts.get(class_name) {
342            if self.is_little_endian {
343                self.virtual_fixups.write_u32::<LittleEndian>(virtual_src)?; // src
344                self.virtual_fixups
345                    .write_i32::<LittleEndian>(self.contents_class_name_section_index)?; // dst_section_index, `__classnames__` section is 0
346                self.virtual_fixups
347                    .write_u32::<LittleEndian>(*class_name_offset)?; // dst(virtual_fixup.dst)
348            } else {
349                self.virtual_fixups.write_u32::<BigEndian>(virtual_src)?; // src
350                self.virtual_fixups
351                    .write_i32::<BigEndian>(self.contents_class_name_section_index)?; // dst_section_index, `__classnames__` section is 0
352                self.virtual_fixups
353                    .write_u32::<BigEndian>(*class_name_offset)?; // dst(virtual_fixup.dst)
354            };
355            Ok(())
356        } else {
357            MissingClassInClassnamesSectionSnafu { class_name }.fail()
358        }
359    }
360
361    /// Write all(`local`, `global` and `virtual`) fixups of data section.
362    ///
363    /// # Returns
364    /// (`local_offset`, `global_offset`, `virtual_offset`)
365    fn write_data_fixups(&mut self) -> Result<(u32, u32, u32)> {
366        let local_offset = self.relative_position()?;
367        self.output.write_all(&self.local_fixups)?;
368        self.output.align(16, 0xff)?;
369
370        #[cfg(feature = "tracing")]
371        tracing::debug!(
372            "[global_fixups pointers]\nsrc: {:#?},\ndest(same as virtual.src): {:#?}",
373            self.global_fixups_ptr_src,
374            self.virtual_fixups_ptr_src
375        );
376        let global_offset = self.relative_position()?;
377        self.write_global_fixups()?;
378        self.output.align(16, 0xff)?;
379
380        let virtual_offset = self.relative_position()?;
381        self.output.write_all(&self.virtual_fixups)?;
382        self.output.align(16, 0xff)?;
383        Ok((local_offset, global_offset, virtual_offset))
384    }
385
386    /// Write all section headers.
387    #[inline]
388    fn write_section_headers<B>(
389        &mut self,
390        section_offset: i16,
391        types_header_start: u64,
392        data_header_start: u64,
393        data_section_header: &SectionHeader,
394    ) -> Result<()>
395    where
396        B: ByteOrder,
397    {
398        // `__classnames__` header`
399        SectionHeader::write_classnames::<B>(
400            &mut self.output,
401            section_offset,
402            self.abs_data_offset,
403        )?;
404        // `__types__` header`
405        self.output.set_position(types_header_start);
406        SectionHeader::write_types::<B>(&mut self.output, self.abs_data_offset)?;
407        // `__data__` header`
408        self.output.set_position(data_header_start);
409        data_section_header.write_bytes::<B>(&mut self.output)?;
410        Ok(())
411    }
412
413    /// Go to relative The data position pointed to by ptr.
414    ///
415    /// Returns
416    /// (
417    ///  The last write position for the pointer type,
418    ///  `__data__` section relative offset) destination position,
419    /// )
420    ///
421    /// # Note
422    /// Always align to 16 bytes when writing to the array.
423    /// Note: If a Struct within hkArray<Struct> contains a StringPtr, it does not necessarily align to 16 bytes.
424    #[inline]
425    fn goto_latest_local_dst(&mut self, need_align16: bool) -> Result<(u64, u32)> {
426        let dest_abs_pos = tri!(
427            self.pointed_pos
428                .last()
429                .copied()
430                .context(NotFoundPointedPositionSnafu)
431        );
432        let dest_abs_pos = if need_align16 {
433            align!(dest_abs_pos, 16_u64)
434        } else {
435            dest_abs_pos
436        };
437        self.output.set_position(dest_abs_pos);
438        let local_dst = tri!(self.relative_position());
439        Ok((dest_abs_pos, local_dst))
440    }
441
442    fn update_last_local_dst(&mut self, new_pos: u64) -> Result<()> {
443        self.pointed_pos
444            .last_mut()
445            .map(|last| *last = new_pos)
446            .context(NotFoundPointedPositionSnafu)?; // Update to serialize the next pointed data.
447        Ok(())
448    }
449
450    /// Write a pair of local_fixups.
451    #[inline]
452    fn write_local_fixup_pair(&mut self, local_src: u32, local_dst: u32) -> Result<()> {
453        #[cfg(feature = "tracing")]
454        {
455            let src_abs = self.abs_data_offset + local_src;
456            let dst_abs = self.abs_data_offset + local_dst;
457            tracing::debug!(
458                "[local_fixup] src({local_src}/abs: {src_abs:#x}), dst({local_dst}/abs: {dst_abs:#x})"
459            );
460        }
461        if self.is_little_endian {
462            self.local_fixups.write_u32::<LittleEndian>(local_src)?;
463            self.local_fixups.write_u32::<LittleEndian>(local_dst)?;
464        } else {
465            self.local_fixups.write_u32::<BigEndian>(local_src)?;
466            self.local_fixups.write_u32::<BigEndian>(local_dst)?;
467        }
468        Ok(())
469    }
470
471    /// Write the internal data pointed to by the pointer of `CString` or `StringPtr`.
472    fn serialize_cow(&mut self, v: &Option<Cow<'_, str>>) -> Result<()> {
473        #[cfg(feature = "tracing")]
474        tracing::trace!("pointed_pos:({:#x?})", self.pointed_pos);
475
476        // Skip if `Option::None`(null pointer).
477        if let Some(v) = v {
478            let rel_ptr_start = self.relative_position()?;
479            tri!(self.serialize_ulong(Ulong::new(0))); // ptr size
480
481            let abs_next_ser_pos = self.output.position();
482            #[cfg(feature = "tracing")]
483            tracing::trace!("Serialize `CString`/`StringPtr` ({abs_next_ser_pos:#x}): (\"{v}\")",);
484
485            // local dst location -----------------------------------------------------
486            {
487                let (_, local_dst) = tri!(self.goto_latest_local_dst(false));
488                tri!(self.write_local_fixup_pair(rel_ptr_start, local_dst));
489            }
490
491            let c_string = StdCString::new(v.as_bytes())?;
492            let _ = self.output.write(c_string.as_bytes_with_nul())?;
493            self.output
494                .zero_fill_align(if self.is_in_str_array { 2 } else { 16 })?;
495
496            tri!(self.update_last_local_dst(self.output.position()));
497            // local dst --------------------------------------------------------------
498
499            self.output.set_position(abs_next_ser_pos); // go back to C++ field serialization position.
500        } else {
501            tri!(self.serialize_ulong(Ulong::new(0))); // null ptr, just write ptr size
502        }
503
504        Ok(())
505    }
506}

508/// Endianness and a common write process that takes into account whether the array is being serialized or not.
509macro_rules! impl_serialize_primitive {
510    ($($serialize_fn:ident, $write:ident => ($ty:ty, $cast_ty:ty, $num:path, $eid:path, $vid:path)),+ $(,)?) => {
511        $(
512            #[inline]
513            fn $serialize_fn(self, v: &$ty) -> Result<Self::Ok> {
514                let v = match v {
515                    $num(n) => *n,
516                    $eid(id) => {
517                        let n = tri!(self.event_id_map.get(id.as_ref()).ok_or_else(||
518                            NotFoundEventIdSnafu {
519                                event_id: id.to_string(),
520                            }.build()
521                        ));
522                        n as $cast_ty
523                    },
524                    $vid(id) => {
525                        let n = tri!(self.variable_id_map.get(id.as_ref()).ok_or_else(||
526                            NotFoundVariableIdSnafu {
527                                variable_id: id.to_string(),
528                            }.build()
529                        ));
530                        n as $cast_ty
531                    },
532                };
533                match self.is_little_endian {
534                    true => self.output.$write::<LittleEndian>(v),
535                    false => self.output.$write::<BigEndian>(v),
536                }?;
537                Ok(())
538            }
539        )*
540    };
541}

543/// Endianness and a common write process that takes into account whether the array is being serialized or not.
544macro_rules! impl_serialize_math {
545    ($method:ident, $value_type:ty) => {
546        fn $method(self, v: &$value_type) -> Result<Self::Ok, Self::Error> {
547            match self.is_little_endian {
548                true => self.output.write(v.to_le_bytes().as_slice()),
549                false => self.output.write(v.to_be_bytes().as_slice()),
550            }?;
551            Ok(())
552        }
553    };

556impl<'a, 'ser> Serializer for &'a mut ByteSerializer<'ser> {
557    type Ok = ();
558    type Error = Error;
559
560    type SerializeSeq = Self;
561    type SerializeStruct = StructSerializer<'a, 'ser>;
562    type SerializeFlags = Self;
563
564    #[inline]
565    fn serialize_void(self, _: ()) -> Result<Self::Ok, Self::Error> {
566        Ok(())
567    }
568
569    #[inline]
570    fn serialize_bool(self, v: bool) -> Result<Self::Ok, Self::Error> {
571        self.serialize_uint8(&U8::Number(v as u8))
572    }
573
574    #[inline]
575    /// Assume that the characters are ASCII characters`c_char`. In that case, i8 is used to fit into 128 characters.
576    fn serialize_char(self, v: char) -> Result<Self::Ok, Self::Error> {
577        self.serialize_int8(&I8::Number(v as i8))
578    }
579
580    #[inline]
581    fn serialize_int8(self, v: &I8) -> Result<Self::Ok, Self::Error> {
582        let v = match v {
583            I8::Number(n) => *n,
584            I8::EventId(event_id) => {
585                let n = self.event_id_map.get(event_id.as_ref()).ok_or_else(|| {
586                    NotFoundEventIdSnafu {
587                        event_id: event_id.to_string(),
588                    }
589                    .build()
590                })?;
591                n as i8
592            }
593            I8::VariableId(variable_id) => {
594                let n = tri!(
595                    self.variable_id_map
596                        .get(variable_id.as_ref())
597                        .ok_or_else(|| {
598                            NotFoundVariableIdSnafu {
599                                variable_id: variable_id.to_string(),
600                            }

611    #[inline]
612    fn serialize_uint8(self, v: &U8) -> Result<Self::Ok, Self::Error> {
613        let v = match v {
614            U8::Number(n) => *n,
615            U8::EventId(event_id) => {
616                let n = self.event_id_map.get(event_id.as_ref()).ok_or_else(|| {
617                    NotFoundEventIdSnafu {
618                        event_id: event_id.to_string(),
619                    }
620                    .build()
621                })?;
622                n as u8
623            }
624            U8::VariableId(variable_id) => {
625                let n = tri!(
626                    self.variable_id_map
627                        .get(variable_id.as_ref())
628                        .ok_or_else(|| {
629                            NotFoundVariableIdSnafu {
630                                variable_id: variable_id.to_string(),
631                            }
632                            .build()
633                        })
634                );
635                n as u8
636            }
637        };
638        self.output.write_u8(v)?;
639        Ok(())
640    }

668    // Register data to be written to global_fixups.
669    // Until the data in virtual_fixups is determined (until all C++ classes are written), temporarily write to `IndexMap` as the value of dst is unknown.
670    fn serialize_pointer(self, ptr: &Pointer) -> Result<Self::Ok, Self::Error> {
671        #[allow(clippy::needless_else)]
672        if !ptr.is_null() {
673            // Write global_fixup src(write start) position.
674            let start = self.relative_position()?;
675            #[cfg(feature = "tracing")]
676            tracing::debug!(
677                "Insert `global_fixup.src`({start}:{start:#x}/abs {:#x}): {ptr}",
678                self.output.position()
679            );
680            self.global_fixups_ptr_src.insert(start, ptr.to_static());
681        } else {
682            #[cfg(feature = "tracing")]
683            tracing::debug!("Skip global_fixup.src writing, because it's null ptr.");
684        };
685        self.serialize_ulong(Ulong::new(0))
686    }

693    /// This is called in the Havok Class array or HashMap, or in a serializer in a field.
694    /// Classes in the field may be inlined, in which case `class_meta` will be [`None`].
695    ///
696    /// # what's `class_meta`?
697    /// If `class_meta` exists, it is when writing an `hkobject` with the `name` attribute in XML.
698    /// This must be written to virtual_fixup so that the constructor can be called.
699    fn serialize_struct(
700        self,
701        name: &'static str,
702        class_meta: Option<(&Pointer, Signature)>,
703        sizes: (u64, u64),
704    ) -> Result<Self::SerializeStruct, Self::Error> {
705        #[allow(clippy::needless_else)]
706        let is_root = if let Some((ptr, _sig)) = class_meta {
707            self.output.zero_fill_align(16)?; // Make sure `virtual_fixup.src`(each Class) is `align16`.
708
709            let virtual_fixup_abs = self.output.position();
710            #[cfg(feature = "tracing")]
711            tracing::debug!(
712                "serialize struct {name}(index = {ptr}, signature = {_sig}, abs_position = {virtual_fixup_abs:#x})"
713            );
714            {
715                let virtual_src = self.relative_position()?;
716                self.write_virtual_fixups_pair(name, virtual_src)?; // Ok, `virtual_fixup` is known.
717                self.virtual_fixups_ptr_src
718                    .insert(ptr.to_static(), virtual_src); // Backup to write `global_fixups`j
719            }
720
721            // The data pointed to by the pointer member(hkCString/hkStringPtr/`hkArray<T>`)
722            // must first be aligned 16 bytes before it is written.
723            let size = if self.is_x86 { sizes.0 } else { sizes.1 };
724            let abs_latest_local_dst = align!(virtual_fixup_abs + size, 16_u64);
725            self.pointed_pos.push(abs_latest_local_dst);
726            true
727        } else {
728            #[cfg(feature = "tracing")]
729            tracing::debug!("serialize struct {name}(A class within a field.)");
730            false
731        };
732
733        Ok(Self::SerializeStruct::new(self, is_root))
734    }
