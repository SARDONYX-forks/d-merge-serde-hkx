1//! Create event_id & variable map from `hkbBehaviorGraphStringData`.
2
3use havok_classes::Classes;
4use serde_hkx::{EventIdMap, VariableIdMap};
5
6/// Create `eventID` & `variableId` maps from `hkbBehaviorGraphStringData` class
7///
8/// # Example
9/// ```no_run
10/// use serde_hkx::bytes::serde::hkx_header::HkxHeader;
11/// use serde_hkx::errors::SerdeHkxError;
12/// use serde_hkx::{ClassMapKey, EventIdMap, VariableIdMap};
13/// use serde_hkx_features::id_maker::crate_maps_from_id_class;
14/// use serde_hkx_features::ClassMap;
15///
16/// fn main() -> Result<(), SerdeHkxError> {
17///     let class_map = ClassMap::new();
18///     let ptr: ClassMapKey = "#0001".into(); // hkbBehaviorGraphStringData pointer
19///     let (event_id_map, variable_id_map) = class_map
20///         .get(&ptr)
21///         .and_then(|class| crate_maps_from_id_class(class))
22///         .unwrap_or_else(|| (EventIdMap::new(), VariableIdMap::new()));
23///
24///     let hkx = serde_hkx::to_bytes_with_maps(
25///         &class_map,
26///         &HkxHeader::new_skyrim_se(),
27///         event_id_map,
28///         variable_id_map,
29///     )?;
30///     std::fs::write("generated.hkx", &hkx)?;
31///     Ok(())
32/// }
33/// ```
34pub fn crate_maps_from_id_class(id_class: &Classes<'_>) -> Option<(EventIdMap, VariableIdMap)> {
35    if let Classes::hkbBehaviorGraphStringData(graph_class) = id_class {
36        let mut event_map = EventIdMap::new();
37        let mut variable_map = VariableIdMap::new();
38        for (id, event_id) in graph_class.m_eventNames.iter().enumerate() {
39            event_map.0.insert(event_id.to_string(), id);
40        }
41
42        for (id, variable_id) in graph_class.m_variableNames.iter().enumerate() {
43            variable_map.0.insert(variable_id.to_string(), id);
44        }
45        return Some((event_map, variable_map));
46    }
47
48    None
49}
