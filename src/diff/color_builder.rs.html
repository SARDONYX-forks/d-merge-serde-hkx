3use std::fmt::{Display, Write};
4use std::ops::Range;
5
6use imara_diff::Sink;
7use imara_diff::intern::{InternedInput, Interner, Token};

9/// A [`Sink`] that creates a textual diff
10/// in the format typically output by git or gnu-diff if the `-u` option is used
11pub struct UnifiedDiffBuilder<'a, W, T>
12where
13    W: Write,
14    T: Display,
15{
16    before: &'a [Token],
17    after: &'a [Token],
18    interner: &'a Interner<T>,
19
20    pos: u32,
21    before_hunk_start: u32,
22    after_hunk_start: u32,
23    before_hunk_len: u32,
24    after_hunk_len: u32,
25
26    buffer: String,
27    dst: W,
28
29    color: bool,
30}
31
32impl<'a, T> UnifiedDiffBuilder<'a, String, T>
33where
34    T: Display,
35{
36    /// Create a new `UnifiedDiffBuilder` for the given `input`,
37    /// that will return a [`String`].
38    pub fn new(input: &'a InternedInput<T>, use_color: bool) -> Self {
39        Self {
40            before_hunk_start: 0,
41            after_hunk_start: 0,
42            before_hunk_len: 0,
43            after_hunk_len: 0,
44            buffer: String::with_capacity(8),
45            dst: String::new(),
46            interner: &input.interner,
47            before: &input.before,
48            after: &input.after,
49            pos: 0,
50            color: use_color,
51        }
52    }
53}
54
55impl<'a, W, T> UnifiedDiffBuilder<'a, W, T>
56where
57    W: Write,
58    T: Display,
59{
60    /// Create a new `UnifiedDiffBuilder` for the given `input`,
61    /// that will writes it output to the provided implementation of [`Write`].
62    #[allow(unused)]
63    pub fn with_writer(input: &'a InternedInput<T>, writer: W) -> Self {
64        Self {
65            before_hunk_start: 0,
66            after_hunk_start: 0,
67            before_hunk_len: 0,
68            after_hunk_len: 0,
69            buffer: String::with_capacity(8),
70            dst: writer,
71            interner: &input.interner,
72            before: &input.before,
73            after: &input.after,
74            pos: 0,
75            color: false,
76        }
77    }
78
79    fn print_tokens(&mut self, tokens: &[Token], prefix: char) {
80        for &token in tokens {
81            let line = format!("{prefix}{}\n", self.interner[token]);
82            if self.color {
83                match prefix {
84                    '-' => write!(self.buffer, "\x1b[31m{line}\x1b[0m").unwrap(),
85                    '+' => write!(self.buffer, "\x1b[32m{line}\x1b[0m").unwrap(),
86                    _ => self.buffer.write_str(&line).unwrap(),
87                }
88            } else {
89                self.buffer.write_str(&line).unwrap();
90            }
91        }
92    }
93
94    fn flush(&mut self) {
95        if self.before_hunk_len == 0 && self.after_hunk_len == 0 {
96            return;
97        }
98
99        let end = (self.pos + 3).min(self.before.len() as u32);
100        self.update_pos(end, end);
101
102        writeln!(
103            &mut self.dst,
104            "@@ -{},{} +{},{} @@",
105            self.before_hunk_start + 1,
106            self.before_hunk_len,
107            self.after_hunk_start + 1,
108            self.after_hunk_len,
109        )
110        .unwrap();
111        write!(&mut self.dst, "{}", &self.buffer).unwrap();
112        self.buffer.clear();
113        self.before_hunk_len = 0;
114        self.after_hunk_len = 0
115    }
116
117    fn update_pos(&mut self, print_to: u32, move_to: u32) {
118        self.print_tokens(&self.before[self.pos as usize..print_to as usize], ' ');
119        let len = print_to - self.pos;
120        self.pos = move_to;
121        self.before_hunk_len += len;
122        self.after_hunk_len += len;
123    }
124}
125
126impl<W, T> Sink for UnifiedDiffBuilder<'_, W, T>
127where
128    W: Write,
129    T: Display,
130{
131    type Out = W;
132
133    fn process_change(&mut self, before: Range<u32>, after: Range<u32>) {
134        if before.start - self.pos > 6 {
135            self.flush();
136            self.pos = before.start - 3;
137            self.before_hunk_start = self.pos;
138            self.after_hunk_start = after.start - 3;
139        }
140        self.update_pos(before.start, before.end);
141        self.before_hunk_len += before.end - before.start;
142        self.after_hunk_len += after.end - after.start;
143        self.print_tokens(
144            &self.before[before.start as usize..before.end as usize],
145            '-',
146        );
147        self.print_tokens(&self.after[after.start as usize..after.end as usize], '+');
148    }
149
150    fn finish(mut self) -> Self::Out {
151        self.flush();
152        self.dst
153    }
154}
