14struct Histogram {
15    token_occurances: Vec<ListHandle>,
16    pool: ListPool,
17}
18
19pub fn diff<S: Sink>(
20    mut before: &[Token],
21    mut after: &[Token],
22    num_tokens: u32,
23    mut sink: S,
24) -> S::Out {
25    let mut histogram = Histogram::new(num_tokens);
26    let prefix = strip_common_prefix(&mut before, &mut after);
27    strip_common_postfix(&mut before, &mut after);
28    histogram.run(before, prefix, after, prefix, &mut sink);
29    sink.finish()
30}
31
32impl Histogram {
33    fn new(num_buckets: u32) -> Histogram {
34        Histogram {
35            token_occurances: vec![ListHandle::default(); num_buckets as usize],
36            pool: ListPool::new(2 * num_buckets),
37        }
38    }
39
40    fn clear(&mut self) {
41        self.pool.clear();
42    }
43
44    fn token_occurances(&self, token: Token) -> &[u32] {
45        self.token_occurances[token.0 as usize].as_slice(&self.pool)
46    }
47
48    fn num_token_occurances(&self, token: Token) -> u32 {
49        self.token_occurances[token.0 as usize].len(&self.pool)
50    }
51
52    fn populate(&mut self, file: &[Token]) {
53        for (i, &token) in file.iter().enumerate() {
54            self.token_occurances[token.0 as usize].push(i as u32, &mut self.pool);
55        }
56    }
57
58    fn run(
59        &mut self,
60        mut before: &[Token],
61        mut before_off: u32,
62        mut after: &[Token],
63        mut after_off: u32,
64        sink: &mut impl Sink,
65    ) {
66        loop {
67            if before.is_empty() {
68                if !after.is_empty() {
69                    sink.process_change(
70                        before_off..before_off,
71                        after_off..after_off + after.len() as u32,
72                    );
73                }
74                return;
75            } else if after.is_empty() {
76                sink.process_change(
77                    before_off..before_off + before.len() as u32,
78                    after_off..after_off,
79                );
80                return;
81            }
82
83            self.populate(before);
84            match find_lcs(before, after, self) {
85                // no lcs was found, that means that file1 and file2 two have nothing in common
86                Some(lcs) if lcs.len == 0 => {
87                    sink.process_change(
88                        before_off..before_off + before.len() as u32,
89                        after_off..after_off + after.len() as u32,
90                    );
91                    return;
92                }
93                Some(lcs) => {
94                    self.run(
95                        &before[..lcs.before_start as usize],
96                        before_off,
97                        &after[..lcs.after_start as usize],
98                        after_off,
99                        sink,
100                    );
101
102                    // this is equivalent to (tail) recursion but implement as a loop for efficeny reasons
103                    let before_end = lcs.before_start + lcs.len;
104                    before = &before[before_end as usize..];
105                    before_off += before_end;
106
107                    let after_end = lcs.after_start + lcs.len;
108                    after = &after[after_end as usize..];
109                    after_off += after_end;
110                }
111                None => {
112                    // we are diffing two extremly large repetitive file
113                    // this is a worst case for histogram diff with O(N^2) performance
114                    // fallback to myers to maintain linear time complxity
115                    myers::diff(
116                        before,
117                        after,
118                        0, // not used by myers
119                        |mut before: Range<u32>, mut after: Range<u32>| {
120                            before.start += before_off;
121                            before.end += before_off;
122                            after.start += after_off;
123                            after.end += after_off;
124                            sink.process_change(before, after)
125                        },
126                        false,
127                    );
128                    return;
129                }
130            }
131        }
132    }
133}
